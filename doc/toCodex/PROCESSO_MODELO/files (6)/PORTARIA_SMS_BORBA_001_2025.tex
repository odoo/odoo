% ── Art. 1 — Gestor ──────────────────────────────────────────────────
\artigo{Designar o(a) servidor(a) \textbf{\gestorNome}, ocupante do cargo de
\gestorCargo, matrícula n.\textsuperscript{o}~\gestorMat, lotado(a) em
\gestorLotacao, para exercer a função de \textbf{GESTOR(A) DA ATA DE REGISTRO
DE PREÇOS n.\textsuperscript{o}~\arpNum/\exercicio}, no âmbito da Secretaria
Municipal de Saúde de Borba/AM.}

\paragrafo{1}{Ao Gestor da ARP competem as seguintes atribuições, nos termos
do \art{117}, \S\,1.\textsuperscript{o}, da Lei n.\textsuperscript{o}~14.133/2021:}

\begin{enumerate}[label=\Roman*.]
  \item Coordenar e supervisionar a execução da \arpRef, garantindo o cumprimento
    de todas as cláusulas e condições pactuadas;

  \item Emitir e controlar as \textbf{Ordens de Fornecimento (OF)}, observando
    o saldo físico-financeiro de cada item registrado;

  \item Manter atualizado o \textbf{painel de controle} da ARP, com registro de
    todas as OF emitidas, quantidades fornecidas, valores empenhados e pagos,
    e saldo remanescente;

  \item Decidir sobre pedidos de \textbf{reequilíbrio econômico-financeiro},
    após instrução pelo Fiscal Técnico, submetendo à autoridade competente
    os casos que superem sua competência;

  \item Propor à autoridade competente a \textbf{prorrogação, cancelamento ou
    rescisão} da ARP, devidamente fundamentados;

  \item Comunicar às Fornecedoras Registradas quaisquer alterações nas condições
    da ARP, inclusive eventuais restrições orçamentárias;

  \item Publicar trimestralmente no PNCP os \textbf{preços praticados}, nos
    termos do \art{82}, \S\,2.\textsuperscript{o}, da Lei n.\textsuperscript{o}
    14.133/2021;

  \item Elaborar e submeter à autoridade competente o \textbf{Relatório
    Trimestral de Gestão} da ARP, contendo o acompanhamento dos indicadores
    de desempenho (KPIs) previstos na Cláusula~\ref*{sec:kpis} da \arpRef;

  \item Comunicar imediatamente ao Pregoeiro e à autoridade competente qualquer
    irregularidade ou inadimplência da Fornecedora Registrada;

  \item Instruir os eventuais processos sancionatórios, em parceria com o
    Fiscal Técnico.
\end{enumerate}

\paragrafo{2}{O Gestor deverá ser substituído nas hipóteses de impedimento,
afastamento ou vacância, mediante nova portaria de designação.}

\vspace{0.5cm}

% ── Art. 2 — Fiscal Técnico ──────────────────────────────────────────
\artigo{Designar o(a) servidor(a) \textbf{\fiscalNome}, ocupante do cargo de
\fiscalCargo, matrícula n.\textsuperscript{o}~\fiscalMat, lotado(a) em
\fiscalLotacao, para exercer a função de \textbf{FISCAL TÉCNICO(A) DA ATA DE
REGISTRO DE PREÇOS n.\textsuperscript{o}~\arpNum/\exercicio}.}

\paragrafo{1}{Ao Fiscal Técnico competem as seguintes atribuições:}

\begin{enumerate}[label=\Roman*.]
  \item Verificar, em cada entrega, a \textbf{conformidade técnica} dos produtos
    com as especificações do Termo de Referência \trRef, incluindo marca,
    modelo, concentração, prazo de validade residual e condições das embalagens;

  \item Confirmar, em tempo real no portal da ANVISA, a \textbf{validade dos
    registros, notificações e autorizações} de funcionamento dos produtos
    dos Lotes 01 a 05;

  \item Emitir o \textbf{Termo de Recebimento Provisório (TRP)}, em até 2 dias
    úteis após a entrega, e o \textbf{Termo de Recebimento Definitivo (TRD)},
    em até 10 dias úteis após o TRP;

  \item Emitir \textbf{Notificações de Não Conformidade (NNC)} e controlar os
    prazos de substituição dos produtos devolvidos;

  \item Monitorar os \textbf{indicadores de desempenho (KPIs)} definidos na
    Cláusula~12 da \arpRef\ e reportar mensalmente ao Gestor;

  \item Registrar e encaminhar ao Gestor toda não conformidade, atraso ou
    recusa de entrega, com a documentação necessária à instrução de eventual
    processo sancionatório;

  \item Analisar as \textbf{Fichas de Dados de Segurança (FISPQ)} dos produtos
    dos Lotes 01 a 05 e verificar sua adequação às normas da ABNT NBR 14.725
    e da NR-32 (MTE);

  \item Elaborar o \textbf{Relatório Mensal de Fiscalização}, juntando ao
    processo \procNum\ em até 5 dias úteis após o encerramento de cada mês.
\end{enumerate}

\paragrafo{2}{O Fiscal Técnico \textbf{não} tem competência para autorizar
alterações nas condições da ARP, emitir Ordens de Fornecimento ou autorizar
pagamentos; tais atos são de competência exclusiva do Gestor e da autoridade
competente.}

\vspace{0.5cm}

% ── Art. 3 — Fiscal Administrativo (Suplente) ────────────────────────
\artigo{Designar o(a) servidor(a) \textbf{\fiscalAdmNome}, ocupante do cargo
de \fiscalAdmCargo, matrícula n.\textsuperscript{o}~\fiscalAdmMat, lotado(a)
em \fiscalAdmLotacao, para exercer a função de \textbf{FISCAL ADMINISTRATIVO(A)
(SUPLENTE) DA ATA DE REGISTRO DE PREÇOS n.\textsuperscript{o}~\arpNum/\exercicio}.}

\paragrafo{1}{Ao Fiscal Administrativo compete:}

\begin{enumerate}[label=\Roman*.]
  \item Substituir o Fiscal Técnico nos casos de impedimento, afastamento legal
    ou vacância, assumindo integralmente suas atribuições;

  \item Verificar a regularidade documental da NF-e e das certidões de
    regularidade fiscal, previdenciária e trabalhista exigidas como condição
    de pagamento;

  \item Instruir os processos de empenho e liquidação de despesa,
    verificando a correspondência entre a NF-e, a OF e o TRD;

  \item Manter o arquivo físico e eletrônico de todos os documentos da ARP
    (OF, TRP, TRD, NNC, NF-e, certidões) pelo prazo legal de guarda.
\end{enumerate}

\vspace{0.5cm}

% ── Art. 4 — Vedações ────────────────────────────────────────────────
\artigo{É vedado ao Gestor e aos Fiscais designados:}

\begin{enumerate}[label=\Roman*.]
  \item Atestar o recebimento de produtos não entregues ou não inspecionados;
  \item Autorizar fornecimento sem a correspondente Ordem de Fornecimento e
    nota de empenho;
  \item Receber, em nome próprio ou de terceiros, qualquer vantagem pecuniária
    ou presente das Fornecedoras Registradas;
  \item Delegar as atribuições de fiscalização a servidor não designado por
    portaria específica.
\end{enumerate}

\vspace{0.5cm}

% ── Art. 5 — Responsabilidade e capacitação ─────────────────────────
\artigo{Os servidores designados por esta Portaria deverão, no prazo de
\textbf{10 (dez) dias úteis} contados de sua publicação:}

\begin{enumerate}[label=\Roman*.]
  \item Tomar ciência integral do conteúdo do Edital \editalRef, do Termo de
    Referência \trRef\ e da \arpRef;
  \item Assinar o \textbf{Termo de Ciência e Responsabilidade} (Anexo~I desta
    Portaria), a ser juntado ao processo \procNum;
  \item Participar de capacitação interna sobre gestão e fiscalização de ARP
    sob a Lei n.\textsuperscript{o}~14.133/2021, caso ainda não a tenham
    realizado.
\end{enumerate}

\vspace{0.5cm}

% ── Art. 6 — Vigência ────────────────────────────────────────────────
\artigo{Esta Portaria entra em vigor na data de sua publicação no Diário Oficial
do Município de Borba, com efeitos a partir da data de publicação da \arpRef\ no
PNCP (\pubData), e vigorará enquanto perdurar a vigência daquela Ata, incluídas
eventuais prorrogações.}

\vspace{0.5cm}

% ── Art. 7 — Revogação e publicidade ─────────────────────────────────
\artigo{Revogam-se as disposições em contrário. Publique-se no Diário Oficial
do Município de Borba e encaminhe-se cópia ao processo \procNum.}
