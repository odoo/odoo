\section{DO OBJETO}
% ═════════════════════════════════════════════════════════════════════

\begin{legalbox}[\art{82}, \textit{caput}, Lei n.\textsuperscript{o} 14.133/2021]
\small A Ata de Registro de Preços é o \textbf{instrumento vinculante} de
compromisso para futura contratação, com indicação dos fornecedores registrados,
preços, quantidades e condições a serem praticadas nas contratações decorrentes.
\end{legalbox}

\noindent A presente Ata tem por objeto o \textbf{registro de preços} de materiais
de limpeza e higiene para abastecimento da Rede de Atenção à Saúde do Município de
Borba/AM, conforme especificações e quantitativos do Termo de Referência \trRef
(Anexo~I do Edital \editalRef), organizados em \textbf{6 (seis) lotes}, conforme
quadro abaixo:

    automática das obrigações decorrentes desta Ata, ressalvadas as Ordens de
    Fornecimento já emitidas e em andamento.

\begin{enumerate}
  \item O fornecimento dos itens registrados ocorrerá exclusivamente mediante
    emissão de \textbf{Ordens de Fornecimento (OF)}, numeradas sequencialmente e
    assinadas pelo Gestor da ARP;

  \item As OF serão emitidas com antecedência mínima de \textbf{20 dias corridos}
    em relação à data prevista para a entrega, respeitando a logística fluvial
    Manaus--Borba;

  \item A Fornecedora Registrada deverá \textbf{confirmar o recebimento} da OF
    no prazo de \textbf{2 dias úteis}, via sistema eletrônico ou ofício;

  \item O modelo padrão da Ordem de Fornecimento consta do \textbf{Anexo~I}
    desta Ata;

  \item As quantidades solicitadas em cada OF serão calibradas ao ciclo hidrológico
    do Rio Madeira, observando as coberturas mínimas por período:
    \begin{itemize}
      \item \textbf{Período de cheia} (janeiro a junho): cobertura de 30 a 45 dias;
      \item \textbf{Período de estiagem} (julho a outubro): cobertura de 45 a 60
        dias --- Estoque de Reserva Estratégica (ERE);
      \item \textbf{Período de transição} (novembro a dezembro): cobertura de 30 dias;
    \end{itemize}

  \item O intervalo mínimo entre OF para o mesmo item é de \textbf{30 dias},
    salvo emergência documentada;

  \item A OF não poderá superar o saldo registrado na ARP para o respectivo item,
    ficando vedado o fornecimento além do quantitativo máximo registrado.
\end{enumerate}

\subsection{Recebimento Provisório}

O servidor responsável pelo almoxarifado realizará o recebimento provisório no ato
da entrega, verificando a conformidade quantitativa e a integridade das embalagens.
O \textbf{Termo de Recebimento Provisório (TRP)} será emitido em até \textbf{2 dias
úteis} após a entrega.

\subsection{Recebimento Definitivo}

O Fiscal Técnico realizará o recebimento definitivo em até \textbf{10 dias úteis}
após a emissão do TRP, mediante verificação da conformidade qualitativa com as
especificações do TR, confirmação do registro ANVISA e análise da FISPQ.
O \textbf{Termo de Recebimento Definitivo (TRD)} liberará o pagamento.

\subsection{Recusa e Substituição}

Produtos não conformes serão devolvidos via Notificação de Não Conformidade (NNC),
com prazo de \textbf{5 dias úteis} para substituição às expensas da Fornecedora
Registrada, sem prejuízo das penalidades aplicáveis.

    Fornecedora para negociar redução, podendo cancelar o registro em caso de
    recusa (\art{84}, \S\,3.\textsuperscript{o}).

\section{DO CANCELAMENTO DO REGISTRO}
% ═════════════════════════════════════════════════════════════════════

\begin{legalbox}[\art{84}, \S\,2.\textsuperscript{o}, Lei n.\textsuperscript{o} 14.133/2021]
\small
\end{legalbox}

\noindent O registro de preços de uma fornecedora poderá ser cancelado nas seguintes
hipóteses:

\subsection{Por Iniciativa do Órgão Gerenciador}

\begin{enumerate}
  \item A Fornecedora Registrada não cumprir as obrigações desta Ata;
  \item A Fornecedora Registrada não retirar a Ordem de Fornecimento no prazo
    de 5 dias úteis sem justificativa aceita;
  \item Os preços registrados tornarem-se superiores aos praticados no mercado e
    a Fornecedora se recusar a reduzi-los;
  \item A Fornecedora Registrada praticar atos fraudulentos ou sofrer processo
    de falência, recuperação judicial ou extrajudicial;
  \item Por razão de interesse público, mediante ato motivado.
\end{enumerate}

\subsection{Por Iniciativa da Fornecedora Registrada}

\begin{enumerate}
  \item Mediante comprovação de impossibilidade de cumprimento das exigências,
    por fato superveniente, aceita pelo Órgão Gerenciador;
  \item Em caso de recusa de reequilíbrio (\art{84}, \S\,2.\textsuperscript{o},
    IV), sem aplicação de penalidade.
\end{enumerate}

\begin{infobox}[Cancelamento Parcial]
\small
O cancelamento do registro de uma Fornecedora em um lote \textbf{não implica}
o cancelamento da ARP nos demais lotes. O Órgão Gerenciador poderá convocar a
segunda colocada do lote cancelado ou realizar nova licitação restrita ao lote.
\end{infobox}

% ═════════════════════════════════════════════════════════════════════
\section{DOS ÓRGÃOS NÃO PARTICIPANTES (\textit{CARONAS})}
% ═════════════════════════════════════════════════════════════════════

\begin{legalbox}[\art{86}, \S\,4.\textsuperscript{o}, Lei n.\textsuperscript{o} 14.133/2021]
\small
É vedada a adesão à Ata de Registro de Preços por órgão não participante
(\textit{carona}) quando comprometer as quantidades registradas ou onerar o
gerenciamento da Ata.
\end{legalbox}

\begin{enumerate}
  \item Fica \textbf{vedada} a adesão de órgãos não participantes (\textit{caronas})
    a esta ARP sem autorização expressa e motivada do Órgão Gerenciador;

  \item Eventual adesão autorizada ficará limitada ao saldo remanescente de cada
    item e não poderá comprometer o fornecimento ao Órgão Gerenciador;

  \item A adesão, se autorizada, deverá ser formalizada por aditivo à ARP e
    publicada no PNCP.
\end{enumerate}
