% ── Condições desta OF ───────────────────────────────────────────────
\vspace{0.3cm}
{\color{navy}\bfseries\normalsize CONDIÇÕES ESPECÍFICAS DESTA ORDEM DE
  FORNECIMENTO}\\[6pt]
\textcolor{navy!20}{\rule{\linewidth}{0.5pt}}

\begin{enumerate}
  \item \textbf{Prazo:} A entrega deve ser realizada \textbf{até \ofPrazoEntrega},
    correspondendo a \blank{2cm} dias corridos a partir da data de emissão
    desta OF (\ofEmissao), dentro do prazo máximo para o período \ofPeriodo\
    definido na Cláusula~7 da \arpRef;

  \item \textbf{Confirmação:} A Fornecedora deverá confirmar o recebimento desta OF
    em até \textbf{2 (dois) dias úteis}, por e-mail ao Gestor da ARP ou pelo sistema
    eletrônico; o não cumprimento sujeitará a Fornecedora às penalidades da ARP;

  \item \textbf{Documentação na entrega:} Deverão acompanhar os produtos:
    \begin{itemize}
      \item Nota Fiscal Eletrônica (NF-e) em nome da SMS Borba/AM, referenciando
        esta OF (n.\textsuperscript{o}~\ofNum/\exercicio) e a \arpRef;
      \item Ficha de Dados de Segurança --- FISPQ (Lotes 01 a 05, se primeira entrega
        do lote ou após alteração de formulação);
      \item Laudo de equivalência técnica, se o produto ofertado difere da amostra
        aprovada;
    \end{itemize}

  \item \textbf{Validade mínima:} Todos os itens devem ser entregues com prazo de
    validade residual \textbf{$\geq$ 12 meses}, exceto onde especificado de forma
    diversa no TR \trRef;

  \item \textbf{Recebimento provisório} pelo servidor do almoxarifado no ato da
    entrega; \textbf{recebimento definitivo} pelo Fiscal Técnico em até 10 dias
    úteis (art.\,140, Lei n.\textsuperscript{o}~14.133/2021);

  \item \textbf{Faturamento:} A NF-e deverá ser emitida \textbf{somente após}
    a emissão do Termo de Recebimento Definitivo (TRD);

  \item \textbf{Pagamento:} Em até 30 dias corridos contados da entrega da NF-e
    ao setor financeiro da SMS, após o TRD, conforme Cláusula~9 da \arpRef.
\end{enumerate}

\begin{warnbox}[Penalidades por Atraso]
\small
O descumprimento do prazo de entrega sujeita a Fornecedora à multa de mora de
\textbf{0,5\% ao dia} sobre o valor desta OF, até o 30.\textsuperscript{o} dia;
após o 30.\textsuperscript{o} dia, multa compensatória de 10\% do valor total da
ARP e rescisão, conforme Cláusula~13 da \arpRef\ e \art{162} da Lei
n.\textsuperscript{o}~14.133/2021.
\end{warnbox}

\vspace{0.5cm}
{\color{navy}\bfseries\normalsize 1. QUANDO EMITIR}\\[6pt]
\textcolor{navy!20}{\rule{\linewidth}{0.4pt}}

\begin{enumerate}
  \item A OF deve ser emitida \textbf{antes} do prazo de reposição, ou seja,
    quando o estoque atual atingir o \textbf{Ponto de Pedido (PP)}, calculado
    conforme o ETP \blank{2cm} (Seção~2, Tabela de Modelagem);

  \item O Gestor deve checar quinzenalmente a cobertura do Estoque de Segurança
    (ES) de cada item Classe A e emitir a OF assim que o estoque atingir o PP;

  \item O intervalo mínimo entre OF para o mesmo item é de \textbf{30 dias
    corridos}, salvo emergência documentada;

  \item A OF deve ser emitida com \textbf{antecedência mínima de 20 dias corridos}
    em relação à data de entrega esperada, respeitando os prazos logísticos do
    Rio Madeira.
\end{enumerate}

\vspace{0.4cm}

\textcolor{navy!20}{\rule{\linewidth}{0.4pt}}

\vspace{0.4cm}
{\color{navy}\bfseries\normalsize 4. CONTROLE DE SALDO}\\[6pt]
\textcolor{navy!20}{\rule{\linewidth}{0.4pt}}

\begin{enumerate}
  \item O Gestor manterá painel atualizado com o \textbf{saldo físico e financeiro}
    de cada item da ARP, discriminado por lote;

  \item É vedada a emissão de OF que ultrapasse o \textbf{quantitativo máximo
    anual registrado} por item na \arpRef\ (Tabela~da ARP);

  \item Quando o saldo remanescente for inferior a 20\% do total registrado,
    o Gestor deverá avaliar a conveniência de \textbf{prorrogação da ARP} e
    encaminhar manifestação à autoridade competente;

  \item Ordens de Fornecimento que individualmente representem valor inferior
    a \textbf{R\$\,5.000,00} somente serão emitidas em caráter emergencial
    documentado, para garantir viabilidade do frete fluvial.
\end{enumerate}
